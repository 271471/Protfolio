<script>
let adj = ["Dark","Pixel","Shadow","Cyber","Ultra"];
let noun = ["Wolf","Knight","Gamer","Dragon","Ninja"];

function generate() {
  let a = adj[Math.floor(Math.random()*adj.length)];
  let n = noun[Math.floor(Math.random()*noun.length)];
  let num = Math.floor(Math.random()*100);

  document.getElementById("name").textContent = a + n + num;
}
</script>
