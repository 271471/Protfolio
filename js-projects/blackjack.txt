let cards = [];
let sum = 0;

function randomCard() {
  return Math.floor(Math.random() * 11) + 1;
}

function start() {
  cards = [randomCard(), randomCard()];
  update();
}

function hit() {
  cards.push(randomCard());
  update();
}

function update() {
  sum = cards.reduce((a, b) => a + b, 0);

  console.log("Cards:", cards);
  console.log("Sum:", sum);

  if (sum === 21) {
    console.log("Blackjack! You win.");
  } else if (sum > 21) {
    console.log("You busted!");
  }
}
