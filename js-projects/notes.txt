let notes = [];

function addNote(note) {
  notes.push(note);
  console.log("Note added:", note);
}

function showNotes() {
  console.log("All Notes:");

  for (let i = 0; i < notes.length; i++) {
    console.log((i + 1) + ". " + notes[i]);
  }
}

// Example usage:
addNote("Study JavaScript");
addNote("Finish portfolio");
addNote("Review math");

showNotes();
