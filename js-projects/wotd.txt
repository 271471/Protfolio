let words = [
  { word: "Serendipity", definition: "Finding something good without looking for it" },
  { word: "Resilience", definition: "Ability to recover from difficulties" },
  { word: "Algorithm", definition: "Step-by-step process for solving a problem" },
  { word: "Innovation", definition: "Creating new ideas or methods" },
  { word: "Logic", definition: "Reasoning used to reach conclusions" }
];

function showWord() {
  let randomIndex = Math.floor(Math.random() * words.length);
  let selected = words[randomIndex];

  console.log("Word of the Day:", selected.word);
  console.log("Definition:", selected.definition);
}

// Example usage:
showWord();
