let score = 0;

function question(promptText, answer) {
  let user = Number(prompt(promptText));

  if (user === answer) {
    score++;
  }
}

question("5 + 3 =", 8);
question("10 + 4 =", 14);
question("6 + 7 =", 13);
question("9 + 2 =", 11);
question("8 + 5 =", 13);

function grade() {
  console.log("Score:", score, "/5");

  if (score === 5) console.log("Grade: A");
  else if (score === 4) console.log("Grade: B");
  else if (score === 3) console.log("Grade: C");
  else if (score === 2) console.log("Grade: D");
  else console.log("Grade: F");
}

grade();
