<script>
function updateClock() {
  let now = new Date();

  let options = {
    timeZone: "America/New_York",
    hour: "2-digit",
    minute: "2-digit",
    second: "2-digit"
  };

  document.getElementById("clock").textContent =
    new Intl.DateTimeFormat("en-US", options).format(now);
}

setInterval(updateClock, 1000);
updateClock();
</script>
