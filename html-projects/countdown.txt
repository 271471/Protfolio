<h2 id="time">10</h2>

<button onclick="start()">Start</button>

<script>
let time = 10;
let interval;

function start() {
  interval = setInterval(() => {
    time--;
    document.getElementById("time").textContent = time;

    if (time <= 0) {
      clearInterval(interval);
      document.getElementById("time").textContent = "Done!";
    }
  }, 1000);
}
</script>
