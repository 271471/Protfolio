tasks = []

while True:
    print("\nTo-Do List")
    print("1. Add task")
    print("2. View tasks")
    print("3. Remove task")
    print("4. Exit")

    choice = input("Choose an option: ")

    if choice == "1":
        task = input("Enter a task: ")
        tasks.append(task)
        print("Task added.")

    elif choice == "2":
        print("\nYour Tasks:")
        if len(tasks) == 0:
            print("No tasks yet.")
        else:
            for i, task in enumerate(tasks):
                print(i + 1, task)

    elif choice == "3":
        for i, task in enumerate(tasks):
            print(i + 1, task)

        remove = int(input("Enter task number to remove: ")) - 1

        if 0 <= remove < len(tasks):
            tasks.pop(remove)
            print("Task removed.")
        else:
            print("Invalid number.")

    elif choice == "4":
        print("Goodbye!")
        break

    else:
        print("Invalid choice")
